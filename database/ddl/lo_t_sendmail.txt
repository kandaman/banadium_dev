CREATE TABLE lo_t_sendmail
(
  ticket_id character varying(25) NOT NULL,
  mail_group character varying(20) NOT NULL,
  user_cd character varying(100) NOT NULL,
  sakujo_flg character varying(20),
  touroku_sha character varying(100),
  touroku_bi timestamp(6) without time zone,
  koushin_sha character varying(100),
  koushin_bi timestamp(6) without time zone,
  primary key (ticket_id, mail_group, user_cd)
)
