create table lo_t_kikaku_shohin_file(
kikaku_id  varchar(20)
,kikaku_edaban smallint
,file_no smallint
,file_name varchar(512)
,file_path varchar(512)
,comment varchar(1000)
,sakujo_flg  varchar(20)
,touroku_sha  varchar(100)
,touroku_bi  timestamp(6) without time zone
,koushin_sha  varchar(100)
,koushin_bi  timestamp(6) without time zone
,primary key (kikaku_id, kikaku_edaban, file_no)
);
